function alphabetPosition(text) {
  const alphabet = 'abcdefghijklmnopqrstuvwxyz'
  let result = []
  const textArr = text.replace(/[^a-zA-Z]+/g, "").toLowerCase().split('')
  textArr.forEach(e => {
  return result.push(alphabet.indexOf(e) + 1)
})
  text = result.join(' ')
  return text;
}
